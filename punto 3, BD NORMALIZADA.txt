select distinct(cliente.dniCliente),nombreApellidoCliente
from cliente inner join reparacion on cliente.dniCliente=reparacion.dniCliente
where cliente.dniCliente not in ( select cliente.dnicliente
									from cliente inner join reparacion on cliente.dniCliente=reparacion.dniCliente
									where cliente.tarjetaSecundaria=reparacion.tarjetaReparacion)
order by cliente.dniCliente asc
